async function submitToCalendar(booking) { setStatus('submitting', '⏳ Submitting appointment to Google Calendar…'); let calendarOk = false; try { const res = await fetch('/api/calendar', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ booking }) }); const data = await res.json(); calendarOk = !!data.success; } catch { calendarOk = false; } // Fire the email regardless of calendar outcome — owner still wants the notification, // and the customer still wants their confirmation. Failures here don't block the UI. let emailOk = false; try { const res = await fetch('/api/email', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ booking }) }); const data = await res.json(); emailOk = !!data.success; } catch { emailOk = false; } if (calendarOk && emailOk) { setStatus('success', '✅ Appointment submitted to Google Calendar — confirmation email sent.'); } else if (calendarOk && !emailOk) { setStatus('success', '✅ Appointment submitted to Google Calendar. (Confirmation email could not be sent — please follow up directly.)'); } else if (!calendarOk && emailOk) { setStatus('manual', '⚠️ Confirmation email sent, but please add this appointment to your calendar manually using the booking summary above.'); } else { setStatus('manual', '⚠️ Please add this appointment to your calendar manually using the booking summary above.'); } }
